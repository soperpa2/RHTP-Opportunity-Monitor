from __future__ import annotations from fastapi import FastAPI, Form, Request from fastapi.responses import HTMLResponse, RedirectResponse, JSONResponse from fastapi.staticfiles import StaticFiles from fastapi.templating import Jinja2Templates from .database import init_schema from .repository import dashboard_rows, list_sources, seed_sources, stats, update_review from .worker import run_crawl app = FastAPI(title="RHTP Opportunity Monitor", version="1.0.0") app.mount("/static", StaticFiles(directory="app/static"), name="static") templates = Jinja2Templates(directory="app/templates") @app.get("/health") def health(): return {"ok": True} @app.get("/", response_class=HTMLResponse) def dashboard(request: Request, include: str | None = None, state: str | None = None): return templates.TemplateResponse("dashboard.html", { "request": request, "rows": dashboard_rows(include=include, state=state), "sources": list_sources(active_only=False), "stats": stats(), "include": include or "", "state": state or "", }) @app.get("/api/opportunities") def api_opportunities(include: str | None = None, state: str | None = None): return dashboard_rows(include=include, state=state) @app.get("/api/sources") def api_sources(): return list_sources(active_only=False) @app.post("/review/{opportunity_id}") def review(opportunity_id: str, review_status: str = Form(...), pursue_status: str = Form(""), notes: str = Form("")): update_review(opportunity_id, review_status, pursue_status or None, notes or None) return RedirectResponse("/", status_code=303) @app.post("/admin/init") def admin_init(): init_schema() seeded = seed_sources() return {"initialized": True, "seeded_sources": seeded} @app.post("/admin/crawl") def admin_crawl(limit: int | None = None): return JSONResponse(run_crawl(limit=limit))
